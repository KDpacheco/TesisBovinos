CREATE TABLE public.animal (
	animal_id varchar(255) NOT NULL,
	animal_madre varchar(255) NULL,
	animal_padre varchar(255) NULL,
	animal_color varchar(255) NOT NULL,
	animal_peso numeric(10) NOT NULL,
	animal_raza int4 NOT NULL,
	animal_sexo varchar(255) NOT NULL,
	animal_categoria int4 NOT NULL,
	animal_nacimiento date NOT NULL,
	animal_imagen varchar(255) NULL,
	animal_estado int4 NOT NULL,
	deleted_at timestamp(0) NULL,
	created_at timestamp(0) NULL,
	updated_at timestamp(0) NULL,
	codigo_bien varchar NULL,
	animal_produccion int4 NULL,
	animal_abierto bool NOT NULL,
	fecha_secado timestamp NULL,
	animal_arete numeric NULL,
	CONSTRAINT animal_pkey PRIMARY KEY (animal_id),
	CONSTRAINT animal_fk FOREIGN KEY (animal_categoria) REFERENCES public.categoria(categoria_id),
	CONSTRAINT animal_fk_estado FOREIGN KEY (animal_estado) REFERENCES public.estados(estados_id),
	CONSTRAINT animal_fk_produccion FOREIGN KEY (animal_produccion) REFERENCES public.produccion(produccion_id),
	CONSTRAINT animal_fk_raza FOREIGN KEY (animal_raza) REFERENCES public.raza(raza_id)
);
